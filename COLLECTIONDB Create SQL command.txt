

    CREATE TABLE "COLLECTION_DB"

        (

        "COLLECTION_ID" INT not null primary key
                GENERATED ALWAYS AS IDENTITY
                (START WITH 1, INCREMENT BY 1),   
		"COLLECTION_UUID" VARCHAR(250),
		"COLLECTION_NAME" VARCHAR(250),
        "COLLECTION_TYPE" VARCHAR(250),	
		"COLLECTION_NOTE" VARCHAR(500),
		"USER_UUID" VARCHAR(250),
		CONSTRAINT uc_CollectionID UNIQUE (COLLECTION_UUID,COLLECTION_NAME)
        );
