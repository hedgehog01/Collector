

    CREATE TABLE "COLLECTIONDB"

        (

        "COLLECTION_ID" INT not null primary key
                GENERATED ALWAYS AS IDENTITY
                (START WITH 1, INCREMENT BY 1),   
		"USER_EMAIL" VARCHAR(150),
		"COLLECTION_UUID" VARCHAR(150),
		"COLLECTION_NAME" VARCHAR(150),
        "COLLECTION_TYPE" VARCHAR(150),
		"COLLECTION_Description" VARCHAR(500)       
        );
